Simple Calculator


import tkinter as tk

def update_result(event):
    try:
        result = eval(entry.get())
        result_label.config(text=str(result))
    except:
        result_label.config(text="Error")

# Create the main window
window = tk.Tk()
window.title("AI Project Calculator")

# Create the entry field
entry = tk.Entry(window, width=25)
entry.grid(row=0, column=0, columnspan=4)
entry.bind("", update_result)

# Create the result label
result_label = tk.Label(window, width=30, anchor="e")
result_label.grid(row=1, column=0, columnspan=4)

# Create the number buttons
buttons = [
    ("7", 2, 0), ("8", 2, 1), ("9", 2, 2), ("/", 2, 3),
    ("4", 3, 0), ("5", 3, 1), ("6", 3, 2), ("*", 3, 3),
    ("1", 4, 0), ("2", 4, 1), ("3", 4, 2), ("-", 4, 3),
    ("0", 5, 0), (".", 5, 1), ("+", 5, 3), 
]

for button in buttons:
    text, row, col = button
    tk.Button(window, text=text, width=5, command=lambda text=text: entry.insert(tk.END, text)).grid(row=row, column=col)

# Run the main event loop
window.mainloop()

Bill Generator


import tkinter as tk

def generate_bill():
    bill_text.delete(1.0, tk.END)
    total = 0

    # Collect item details
    items = []
    for item_entry, price_entry, quantity_entry in zip(item_entries, price_entries, quantity_entries):
        try:
            item_name = item_entry.get()
            item_price = float(price_entry.get())
            item_quantity = int(quantity_entry.get())
            items.append((item_name, item_price, item_quantity))
        except ValueError:
            pass

    # Generate the bill
    bill_text.insert(tk.END, "************ Bill ************\n")
    bill_text.insert(tk.END, "Item\t\tPrice\t\tQuantity\tTotal\n")

    for item in items:
        item_name, item_price, item_quantity = item
        item_total = item_price * item_quantity
        total += item_total

        bill_text.insert(tk.END, f"{item_name}\t\t₹{item_price:.2f}\t\t{item_quantity}\t\t\t₹{item_total:.2f}\n")

    bill_text.insert(tk.END, "-------------------------------\n")
    bill_text.insert(tk.END, f"Total:\t\t\t\t\t\t₹{total:.2f}\n")
    bill_text.insert(tk.END, "*******************************\n")

# Create the main window
window = tk.Tk()
window.title("Shop Bill Generator")

# Create empty lists to store the entry widgets
item_entries = []
price_entries = []
quantity_entries = []

# Function to add a new item row
def add_item_row():
    item_label = tk.Label(window, text="Item Name:")
    item_label.pack()
    item_entry = tk.Entry(window)
    item_entry.pack()
    item_entries.append(item_entry)

    price_label = tk.Label(window, text="Price (₹):")
    price_label.pack()
    price_entry = tk.Entry(window)
    price_entry.pack()
    price_entries.append(price_entry)

    quantity_label = tk.Label(window, text="Quantity:")
    quantity_label.pack()
    quantity_entry = tk.Entry(window)
    quantity_entry.pack()
    quantity_entries.append(quantity_entry)

# Create the "Add Item" button
add_item_button = tk.Button(window, text="Add Item", command=add_item_row)
add_item_button.pack()

# Create the "Generate Bill" button
generate_button = tk.Button(window, text="Generate Bill", command=generate_bill)
generate_button.pack()

# Create the text area for the bill
bill_text = tk.Text(window, height=10, width=40)
bill_text.pack()

# Start the main event loop
window.mainloop()
